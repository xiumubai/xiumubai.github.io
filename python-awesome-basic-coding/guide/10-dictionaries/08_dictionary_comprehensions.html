class LazyDict:
    """å»¶è¿Ÿè®¡ç®—å­—å…¸"""
    def __init__(self, generator_func):
        self._generator_func = generator_func
        self._cache = {}
    
    def __getitem__(self, key):
        if key not in self._cache:
            # åªåœ¨éœ€è¦æ—¶è®¡ç®—

        return self._cache[key]
    
    def keys(self):
        return (k for k, v in self._generator_func())
