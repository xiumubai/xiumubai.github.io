class BorrowRecord:
    """å€Ÿé˜…è®°å½•ç±»"""
    
    def __init__(self, reader, book):
        """åˆå§‹åŒ–å€Ÿé˜…è®°å½•"""
        self.reader = reader
        self.book = book
        self.borrow_date = datetime.now()
        self.due_date = self._calculate_due_date()
        self.return_date = None
        self.fine_paid = 0.0
    
    def _calculate_due_date(self):
        """è®¡ç®—åº”è¿˜æ—¥æœŸ"""
        max_days = Reader._borrow_limits[self.reader.reader_type]["max_days"]
        return self.borrow_date + timedelta(days=max_days)
    
    def is_overdue(self):
        """æ£€æŸ¥æ˜¯å¦é€¾æœŸ"""
        if self.return_date:  # å·²å½’è¿˜
            return False

    def days_overdue(self):
        """è®¡ç®—é€¾æœŸå¤©æ•°"""
        if not self.is_overdue():
            return 0
        
        check_date = self.return_date or datetime.now()
        return (check_date - self.due_date).days
    
    def calculate_fine(self):
        """è®¡ç®—ç½šé‡‘"""
        overdue_days = self.days_overdue()
        if overdue_days <= 0:
            return 0.0
        
        # ç½šé‡‘è§„åˆ™ï¼šæ¯å¤©0.5å…ƒ
        fine_per_day = 0.5
        return overdue_days * fine_per_day
    
    def get_record_info(self):
        """è·å–è®°å½•ä¿¡æ¯"""
        return {
            "è¯»è€…": self.reader.name,
            "å›¾ä¹¦": self.book.title,
            "å€Ÿé˜…æ—¥æœŸ": self.borrow_date.strftime("%Y-%m-%d %H:%M"),
            "åº”è¿˜æ—¥æœŸ": self.due_date.strftime("%Y-%m-%d"),
            "å½’è¿˜æ—¥æœŸ": self.return_date.strftime("%Y-%m-%d %H:%M") if self.return_date else "æœªå½’è¿˜",
            "æ˜¯å¦é€¾æœŸ": self.is_overdue(),
            "é€¾æœŸå¤©æ•°": self.days_overdue(),
            "åº”ç¼´ç½šé‡‘": self.calculate_fine()
