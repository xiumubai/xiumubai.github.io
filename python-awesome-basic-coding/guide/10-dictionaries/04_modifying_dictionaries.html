class ConfigManager:
    def __init__(self, initial_config=None):
        self.config = initial_config or {}
        self.history = [self.config.copy()]
    
    def set(self, key_path, value):
        """
        ä½¿ç”¨ç‚¹åˆ†éš”è·¯å¾„è®¾ç½®é…ç½®å€¼

        # åˆ›å»ºåµŒå¥—ç»“æ„

            target = target[key]
        
        # è®¾ç½®æœ€ç»ˆå€¼
        target[keys[-1]] = value
        self.history.append(self.config.copy())
    
    def update_section(self, section_path, updates):
        """
        æ›´æ–°é…ç½®æ®µ
        """
        keys = section_path.split('.') if section_path else []
        target = self.config
        
        for key in keys:
            if key not in target:
                target[key] = {}
            target = target[key]
        
        target.update(updates)
        self.history.append(self.config.copy())
    
    def remove(self, key_path):
        """
        åˆ é™¤é…ç½®é¡¹
        """
        keys = key_path.split('.')
        target = self.config
        
        # å¯¼èˆªåˆ°çˆ¶çº§
        for key in keys[:-1]:
            if key not in target:
                return False
            target = target[key]
        
        # åˆ é™¤æœ€ç»ˆé”®
        if keys[-1] in target:
            del target[keys[-1]]
            self.history.append(self.config.copy())
            return True
        return False
    
    def rollback(self):
        """
        å›æ»šåˆ°ä¸Šä¸€ä¸ªçŠ¶æ€
        """
        if len(self.history) > 1:
            self.history.pop()  # ç§»é™¤å½“å‰çŠ¶æ€
            self.config = self.history[-1].copy()
            return True
        return False
    
    def get_config(self):
        return self.config.copy()
