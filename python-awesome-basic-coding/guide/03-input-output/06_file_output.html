def safe_write_file(filename, content, encoding='utf-8', backup=True):
    """

    try:

            print(f"å·²åˆ›å»ºå¤‡ä»½æ–‡ä»¶ï¼š{backup_name}")
        
        # å†™å…¥æ–°å†…å®¹

        print(f"é”™è¯¯ï¼šå†™å…¥æ–‡ä»¶æ—¶å‘ç”Ÿç³»ç»Ÿé”™è¯¯ - {e}")
