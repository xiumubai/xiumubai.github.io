def binary_search(arr, target):
    """äºŒåˆ†æŸ¥æ‰¾ç®—æ³•"""
    left, right = 0, len(arr) - 1
    
    while left <= right:

            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return -1  # æœªæ‰¾åˆ°

    operators = []
    i = 0
    
    def precedence(op):
        if op in ['+', '-']:
            return 1
        if op in ['*', '/']:
            return 2
        return 0
    
    def apply_operator():
        if len(numbers) < 2 or not operators:
            return
        
        b = numbers.pop()
        a = numbers.pop()
        op = operators.pop()
        
        if op == '+':
            numbers.append(a + b)
        elif op == '-':
            numbers.append(a - b)
        elif op == '*':
            numbers.append(a * b)
        elif op == '/':
            numbers.append(a / b)

        char = expression[i]
        
        if char.isdigit():
            # è¯»å–å®Œæ•´çš„æ•°å­—
            num = 0
            while i < len(expression) and expression[i].isdigit():

    while operators:
        apply_operator()
