    def create_state_machine(initial_state="idle"):
        """åˆ›å»ºä¸€ä¸ªçŠ¶æ€æœº"""

        state_history = [initial_state]
        
        def get_state():
            return current_state
        
        def set_state(new_state):
            nonlocal current_state
            if new_state != current_state:

                current_state = new_state

                print(f"çŠ¶æ€å˜åŒ–: {old_state} -> {new_state}")

            transitions = {

                "error": ["idle"]

            state_history.clear()
            state_history.append(initial_state)
            print(f"çŠ¶æ€æœºå·²é‡ç½®ä¸º: {initial_state}")
        
        return get_state, transition_to, get_history, reset
    
    # åˆ›å»ºçŠ¶æ€æœº
